Reflection

import "reflect"

func Swap(array []interface{}, i, j int) {
	var temp reflect.Value
	arrayValue := reflect.ValueOf(array)
	temp.Set(arrayValue.Index(i))
	arrayValue.Index(i).Set(arrayValue.Index(j))
	arrayValue.Index(j).Set(temp)
}
